      type: 'price_check',
      product: 'hotel',

      results.flights.result.price +

      results.activities.result.price;

    const savings = trip.budget - total;

      breakdown: results,
      total,
      budget: trip.budget,
      savings,
      confidence: Math.min(...Object.values(results).map(r => r.confidence))
