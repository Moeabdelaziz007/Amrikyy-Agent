// packages/shared/aix-core/src/AgentRegistry.ts

interface RegisteredAgent {
  id: string;
  name: string;
  config: AIXConfig;
  idCard: IDCard;
  status: 'active' | 'inactive' | 'maintenance';
  loaded_at: Date;
  last_activity: Date;
  total_tasks_completed: number;
  average_response_time: number;
  user_rating: number;
}

class AgentRegistry {
  private agents: Map<string, RegisteredAgent> = new Map();
  
  async register(name: string, data: Partial<RegisteredAgent>): Promise<void> {
    this.agents.set(name, {
      id: this.generateId(),
      name,
      status: 'active',
      loaded_at: new Date(),
      last_activity: new Date(),
      total_tasks_completed: 0,
      average_response_time: 0,
      user_rating: 0,
      ...data
    } as RegisteredAgent);
  }
  
  async getAll(): Promise<RegisteredAgent[]> {
    return Array.from(this.agents.values());
  }
  
  async getByName(name: string): Promise<RegisteredAgent | undefined> {
    return this.agents.get(name);
  }
  
  async updateActivity(name: string): Promise<void> {
    const agent = this.agents.get(name);
    if (agent) {
      agent.last_activity = new Date();
      agent.total_tasks_completed++;
    }
  }
}
